C++11 random engine

Example - sleep for 100-1000 milliseconds

#include <random>
#include <thread>
#include <chrono>

std::random_device rd;
std::default_random_engine dre(rd());
std::uniform_int_distribution<int> id(100,1000);

std::this_thread::sleep_for(std::chrono::milliseconds(id(dre)));
